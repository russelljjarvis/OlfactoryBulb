            .order_by(CellModel.class_name)

            }

[docs]    def get_odor_gloms(self, odors):
        if odors == 'all':
            return None

        result = set()

        for odor in odors:
            odor_gloms = OdorGlom \
                .select(OdorGlom.glom_id) \
                .distinct() \
                .join(Odor) \
                .where(Odor.name == odor)

            for glom in odor_gloms:
                result.add(glom.glom_id)

        return list(result)
